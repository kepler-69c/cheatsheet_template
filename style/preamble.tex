%% style/preamble.tex %%

% encoding and fonts
\usepackage[utf8]{inputenc} % utf-8 input
\usepackage[T1]{fontenc}    % better output font encoding
\usepackage{lmodern}        % scalable Latin Modern font

% math
\usepackage{mathtools} % loads amsmath
\usepackage{amsthm}
\usepackage{amsfonts}
\usepackage{amssymb}   % common math symbols
\usepackage{mathrsfs}  % script fonts

% tables, colors, graphics
\usepackage{graphicx,overpic}
\usepackage[dvipsnames]{xcolor}
\usepackage{booktabs}    % nicer tables
\usepackage{multirow,multicol}
\usepackage{supertabular,xltabular,makecell}

% lists
\usepackage{enumitem}

% code and highlighting
\usepackage{minted}
\usepackage{soul}        % highlighting (\hl)
\usepackage{csquotes}    % quotation support

% framed boxes and algorithms
\usepackage[most]{tcolorbox}
\usepackage[ruled,linesnumbered]{algorithm2e}

% spacing and geometry
\usepackage{calc,ifthen}
\usepackage{geometry}
\geometry{top=1.2cm,left=1cm,right=1cm,bottom=0.8cm,headsep=10pt}

% hyperlinks
\usepackage{hyperref}

% theorems
\newtheorem{example}[section]{Example}

% math operators
\DeclareMathOperator*{\argmax}{arg\,max}
\DeclareMathOperator*{\argmin}{arg\,min}
\DeclareMathOperator*{\E}{\mathrm{E}}
\DeclareMathOperator*{\Var}{\mathrm{Var}}
\DeclareMathOperator*{\w}{\mathbf{w}}
\DeclareMathOperator*{\W}{\mathbf{W}}
\DeclareMathOperator*{\x}{\mathbf{x}}
\DeclareMathOperator*{\X}{\mathbf{X}}
\DeclareMathOperator*{\y}{\mathbf{y}}
\DeclareMathOperator*{\z}{\mathbf{z}}
\DeclareMathOperator*{\vv}{\mathbf{v}}

\newcommand{\eqqcolon}{\mathrel{\rotatebox[origin=c]{180}{$\coloneqq$}}}

% spacing
\setlength{\parindent}{0pt}
\setlength{\parskip}{0pt plus 0.5ex}
\setlength{\premulticols}{1pt}
\setlength{\postmulticols}{1pt}
\setlength{\multicolsep}{1pt}
\setlength{\columnsep}{2pt}

% section numbering off
\setcounter{secnumdepth}{0}
